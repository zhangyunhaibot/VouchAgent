routes := x402http.RoutesConfig{
    "GET /oracle": {
        Accepts: x402http.PaymentOptions{
            {Scheme: "exact", Price: "$0.001", Network: x402Network, PayTo: cfg.PayeeAddress},
        },
        Description: "Premium RWA oracle data feed (AI-validated gold/BTC/FX with confidence scores)",
        MimeType:    "application/json",
    },
}

// ─────────────────────────────────────────────────────────────────────────────
// examples/server/main.go — handler: return our agent-written oracle snapshot
// ─────────────────────────────────────────────────────────────────────────────
r.GET("/oracle", func(c *ginfw.Context) {
    // 数据快照路径由 ORACLE_STATE_FILE 环境变量指定（指向 web/oracle_state.json）。
    statePath := os.Getenv("ORACLE_STATE_FILE")
    if statePath == "" {
        statePath = "oracle_state.json"
    }
    data, err := os.ReadFile(statePath)
    if err != nil {
        c.JSON(http.StatusOK, ginfw.H{"error": "no oracle data yet — run the agent first"})
        return
    }
    c.Data(http.StatusOK, "application/json", data)
})
